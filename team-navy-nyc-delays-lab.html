3.8 Observed versus Expected On-Time Departure Percentage

# overall on-time rates by airport and hour (pooled)
base_rates <- departures %>%
  group_by(airport, hour) %>%
  summarise(base_on_time = mean(on_time), .groups = "drop")

# airline exposure to each airport-hour
airline_exposure <- departures %>%
  group_by(airline, airport, hour) %>%
  summarise(n = n(), .groups = "drop") %>%
  group_by(airline) %>%
  mutate(weight = n / sum(n)) %>%
  ungroup()

# expected on-time pct for each airline
expected <- airline_exposure %>%
  left_join(base_rates, by = c("airport", "hour")) %>%
  group_by(airline) %>%
  summarise(
    expected_on_time = sum(weight * base_on_time) * 100,
    .groups = "drop"
  )

observed <- departures %>%
  group_by(airline) %>%
  summarise(observed_on_time = mean(on_time) * 100, .groups = "drop")

obs_exp <- observed %>%
  left_join(expected, by = "airline") %>%
  left_join(departures_by_airline %>% select(airline, n), by = "airline")

ggplot(
  obs_exp %>% filter(n >= 1000),
  aes(x = expected_on_time, y = observed_on_time, label = airline)
) +
  geom_point(aes(size = n), alpha = 0.7) +
  geom_abline(intercept = 0, slope = 1, linetype = "dashed") +
